32 std::pair<VectorXd, VectorXd> Bim1D::bernoulli(const VectorXd & x)
33 {
34  VectorXd bp = VectorXd::Zero( x.size() );
35  VectorXd bn = VectorXd::Zero( x.size() );
36 
37  double lim = 1.0e-2;
38 
39  VectorXd ax = x.cwiseAbs();
40 
41  for ( int i = 0; i < x.size(); ++i ) {
42  if ( x(i) == 0.0 ) {
43  bp(i) = 1.0;
44  bn(i) = 1.0;
45  } else if ( ax(i) > 80.0 ) { // Asymptotics.
46  if ( x(i) > 0.0 ) {
47  bp(i) = 0.0 ;
48  bn(i) = x(i);
49  } else {
50  bp(i) = -x(i);
51  bn(i) = 0.0 ;
52  }
53  } else if ( ax(i) <= 800.0 && ax(i) > lim ) { // Intermediate values.
54  bp(i) = x(i) / ( std::exp(x(i)) - 1.0 );
55  bn(i) = x(i) + bp(i);
56  } else { // if ( ax(i) <= lim ). Small values: Taylor expansion.
57  double j = 1.0;
58  double fp = 1.0;
59  double fn = 1.0;
60  double df = 1.0;
61  int segno = 1 ;
62 
63  while ( std::abs(df) > 1.0e-16 ) {
64  j += 1.0;
65  segno = -segno;
66  df *= x(i) / j;
67  fp += df;
68  fn += segno * df;
69  }
70 
71  bp(i) = 1.0 / fp;
72  bn(i) = 1.0 / fn;
73  }
74  }
75 
76  return std::make_pair(bp, bn);
77 }
