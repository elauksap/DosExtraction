172 void DosModel::post_process(const GetPot & config, const std::string & input_experim, std::ostream & output_fitting,
173  std::ostream & output_CV, const VectorXd & x_semic, const VectorXd & dens,
174  const VectorXd & V_simulated, const VectorXd & C_simulated) const
175 {
176  assert( x_semic .size() == dens .size() );
177  assert( V_simulated.size() == C_simulated.size() );
178 
179  CsvParser parser_experim(input_experim, config("hasHeaders", true));
180 
181  double A_semic = parser_experim.importCell(1, 3); // Semiconductor area [m^2].
182  double C_sb = parser_experim.importCell(1, 4); // Sbord capacitance [F].
183 
184  VectorXd V_experim = parser_experim.importCol(1);
185  VectorXd C_experim = parser_experim.importCol(2);
186 
187  assert( V_experim.size() == C_experim.size() );
188 
189  // Sorting "V_experim" and "C_experim". The order is established by "V_experim".
190  {
191  VectorXpair<double> order = numerics::sort_pair( V_experim );
192  VectorXd V_sort = VectorXd::Zero( V_experim.size() );
193  VectorXd C_sort = VectorXd::Zero( C_experim.size() );
194 
195  for ( int i = 0; i < order.size(); ++i ) {
196  V_sort(i) = V_experim( order(i).second );
197  C_sort(i) = C_experim( order(i).second );
198  }
199 
200  V_experim = V_sort;
201  C_experim = C_sort;
202  }
203 
204  VectorXd dC_dV_experim = numerics::deriv(C_experim, V_experim );
205  VectorXd dC_dV_simulated = numerics::deriv(C_simulated.array() * A_semic + C_sb, V_simulated);
206 
207  double charge_center_of_mass = numerics::trapz( x_semic.cwiseProduct(dens) ) / numerics::trapz( dens );
208  double cAccStar = C_simulated.maxCoeff(); // Simulated.
209 
210  unsigned j_e = 0;
211  dC_dV_experim .maxCoeff(&j_e);
212 
213  unsigned j_s = 0;
214  dC_dV_simulated.maxCoeff(&j_s);
215 
216  double V_shift = V_simulated(j_s) - V_experim(j_e);
217 
218  VectorXd C_interp = numerics::interp1(V_experim, C_experim, V_simulated.array() - V_shift);
219  VectorXd dC_dV_interp = numerics::interp1(V_experim, dC_dV_experim, V_simulated.array() - V_shift);
220 
221  double error_L2 = std::sqrt( numerics::error_L2(C_interp, C_simulated.array() * A_semic + C_sb, V_simulated, V_shift) );
222  double error_H1 = std::sqrt( error_L2 * error_L2 +
223  numerics::error_L2(dC_dV_interp, dC_dV_simulated, V_simulated, V_shift)
224  );
225 
226  // Print to output.
227  output_fitting << std::endl;
228  output_fitting << "V_shift = " << V_shift << std::endl;
229  output_fitting << "Charge center of mass = " << charge_center_of_mass << std::endl;
230  output_fitting << "C_acc* = " << cAccStar << std::endl;
231  output_fitting << std::endl;
232  output_fitting << "Distance between experimental and simulated voltage values:" << std::endl;
233  output_fitting << "\t L2-distance = " << error_L2 << std::endl;
234  output_fitting << "\t H1-distance = " << error_H1 << std::endl;
235 
236  output_CV << "V_experim, C_experim, dC/dV_experim, V_simulated, C_simulated, dC/dV_simulated" << std::endl;
237 
238  for ( int i = 0; i < std::max( V_simulated.size(), V_experim.size() ); ++i ) {
239  if ( i < V_experim.size() ) {
240  output_CV << V_experim(i) << ", " << C_experim(i) << ", " << dC_dV_experim(i) << ", ";
241  } else {
242  output_CV << ",,, ";
243  }
244 
245  if ( i < V_simulated.size() ) {
246  output_CV << V_simulated(i) - V_shift << ", " << C_simulated(i) * A_semic + C_sb << ", " << dC_dV_simulated(i);
247  } else {
248  output_CV << ",,";
249  }
250 
251  output_CV << std::endl;
252  }
253 
254  return;
