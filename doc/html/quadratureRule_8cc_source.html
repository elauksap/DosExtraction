36 void GaussHermiteRule::apply_iterative_algorithm(const Index & maxIterationsNo, const Real & tolerance)
37 {
38  Real p1, p2, temp, dp;
39  Real z, zOld;
40 
41  for ( Index i = 0; i < (nNodes_ + 1) / 2; ++i ) { // The roots are symmetric about the origin,
42  // so only finding half of them is needed.
43  // Good guesses at initial values for the most largest roots.
44  if ( i == 0 ) {
45  z = std::sqrt(2.0 * nNodes_ + 1.0) - 1.85575 * std::pow(2.0 * nNodes_ + 1.0, - 0.16667);
46  } else if ( i == 1 ) {
47  z -= 1.14 * std::pow(nNodes_, 0.426) / z;
48  } else if ( i == 2 ) {
49  z = 1.86 * z - 0.86 * nodes_(nNodes_ - 1);
50  } else if ( i == 3 ) {
51  z = 1.91 * z - 0.91 * nodes_(nNodes_ - 2);
52  } else {
53  z = 2.0 * z - nodes_(nNodes_ - i + 1);
54  }
55 
56  Index j = 0;
57 
58  for ( ; j < maxIterationsNo; ++j ) { // Refinement by Newton's method.
59  p1 = PI_M4;
60  p2 = 0.0;
61 
62  for ( Index k = 0; k < nNodes_; ++k ) { // Loop up the recurrence relation to evaluate the Hermite polynomial at "z".
63  temp = p2;
64  p2 = p1;
65  p1 = z * std::sqrt( 2.0 / (k + 1.0) ) * p2 - std::sqrt( k / (k + 1.0) ) * temp;
66  }
67 
68  // "p1" is now the desidered Hermite polynomial. Computing its derivative "dp" is needed.
69  dp = std::sqrt(2.0 * nNodes_) * p2;
70 
71  // Newton step
72  zOld = z;
73  z = zOld - p1 / dp;
74 
75  if ( std::abs(z - zOld) <= tolerance ) {
76  break;
77  }
78  }
79 
80  if ( j >= maxIterationsNo ) {
81  throw std::runtime_error("ERROR: GaussHermiteRule::apply_iterative_algorithm() didn't reach convergence.");
82  }
83 
84  nodes_(i) = -z;
85  nodes_(nNodes_ - i - 1) = z;
86 
87  weights_(i) = 2.0 / (dp * dp);
88  weights_(nNodes_ - i - 1) = weights_(i) ;
89  }

92 }
