55  VectorXr bn = VectorXr::Zero( x.size() );

58

60

62  {

64  {
65  bp(i) = 1.0;
66  bn(i) = 1.0;
67  }
68  else if ( ax(i) > 80.0 ) // Asymptotics.
69  {
70  if ( x(i) > 0.0 )
71  {
72  bp(i) = 0.0 ;
73  bn(i) = x(i);
74  }
75  else

84  bn(i) = x(i) + bp(i);
85  }
86  else // if ( ax(i) <= lim ). Small values: Taylor expansion.
87  {
88  Real j = 1.0;

90  Real fn = 1.0;
91  Real df = 1.0;
92  int segno = 1 ;

95  {

97  segno = -segno;
98  df *= x(i) / j;
99  fp += df;
100  fn += segno * df;
101  }
102

104  bn(i) = 1.0 / fn;
105  }
