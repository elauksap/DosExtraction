27 void GaussHermiteRule::apply_iterative_algorithm(const unsigned & maxIterationsNo, const double & tolerance)
28 {
29  double p1, p2, temp, dp;
30  double z, zOld;
31 
32  for ( unsigned i = 0; i < (nNodes_ + 1) / 2; ++i ) { // The roots are symmetric about the origin,
33  // so only finding half of them is needed.
34  // Good guesses at initial values for the most largest roots.
35  if ( i == 0 ) {
36  z = std::sqrt(2.0 * nNodes_ + 1.0) - 1.85575 * std::pow(2.0 * nNodes_ + 1.0, - 0.16667);
37  } else if ( i == 1 ) {
38  z -= 1.14 * std::pow(nNodes_, 0.426) / z;
39  } else if ( i == 2 ) {
40  z = 1.86 * z - 0.86 * nodes_(nNodes_ - 1);
41  } else if ( i == 3 ) {
42  z = 1.91 * z - 0.91 * nodes_(nNodes_ - 2);
43  } else {
44  z = 2.0 * z - nodes_(nNodes_ - i + 1);
45  }
46 
47  unsigned j = 0;
48 
49  for ( j = 0; j < maxIterationsNo; ++j ) { // Refinement by Newton's method.
50  p1 = PI_M4;
51  p2 = 0.0;
52 
53  for ( unsigned k = 0; k < nNodes_; ++k ) { // Loop up the recurrence relation to evaluate the Hermite polynomial at "z".
54  temp = p2;
55  p2 = p1;
56  p1 = z * std::sqrt( 2.0 / (k + 1.0) ) * p2 - std::sqrt( k / (k + 1.0) ) * temp;
57  }
58 
59  // "p1" is now the desidered Hermite polynomial. Computing its derivative "dp" is needed.
60  dp = std::sqrt(2.0 * nNodes_) * p2;
61 
62  // Newton step
63  zOld = z;
64  z = zOld - p1 / dp;
65 
66  if ( std::abs(z - zOld) <= tolerance ) {
67  break;
68  }
69  }
70 
71  if ( j >= maxIterationsNo ) {
72  throw std::runtime_error("ERROR: GaussHermiteRule::apply_iterative_algorithm() didn't reach convergence.");
73  }
74 
75  nodes_(i) = -z;
76  nodes_(nNodes_ - i - 1) = z;
77 
78  weights_(i) = 2.0 / (dp * dp);
79  weights_(nNodes_ - i - 1) = weights_(i) ;
80  }
81 
82  return;
83 }

85 void GaussHermiteRule::apply_using_eigendecomposition()
86 {
87  if (nNodes_ == 1) {
88  nodes_ .fill(0.0);
89  weights_.fill(PI) ;
90  return;
91  }
92 
93  MatrixXd Jac = MatrixXd::Zero( nNodes_, nNodes_ ); // Jacobi matrix.
94  VectorXd k = VectorXd::LinSpaced( nNodes_, 1, nNodes_);
95  VectorXd v = std::sqrt(0.5) * k.cwiseSqrt();
96 
97  for ( unsigned i = 0; i < Jac.rows() - 1; ++i ) {
98  Jac(i + 1, i ) = v(i); // Sub-diagonal.
99  Jac( i , i + 1) = v(i); // Super-diagonal.
100  }
101 
102  SelfAdjointEigenSolver<MatrixXd> eigSolver(Jac);
103 
104  nodes_ = eigSolver.eigenvalues();
105 
106  MatrixXd EigVec = eigSolver.eigenvectors();
107 
108  VectorXd norm2 = (EigVec.transpose() * EigVec ).diagonal().cwiseSqrt(); // To normalize weights.
109 
110  weights_ = ( SQRT_PI * EigVec.row(0).cwiseProduct(EigVec.row(0)).transpose() ).cwiseQuotient(norm2); // sqrt(pi) = beta0;
111 
112  return;
