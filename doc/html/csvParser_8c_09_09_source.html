19  // Get number of rows.

31 
32  // Check separator.
33  if ( std::find(line_.begin(), line_.end(), ',') != line_.end() ) {
34  separator_ = ',';
35  } else if ( std::find(line_.begin(), line_.end(), '\t') != line_.end() ) {
36  separator_ = '\t';
37  } else if ( std::find(line_.begin(), line_.end(), ':') != line_.end() ) {
38  separator_ = ':';
39  } else if ( std::find(line_.begin(), line_.end(), ' ') != line_.end() ) {
40  separator_ = ' ';
41  } else {
42  throw std::ifstream::failure("ERROR: input file isn't either comma-, TAB-, colon- or space-separated.");
43  }
44 
45  // Get number of columns.
46  {
47  // Import the first row: .csv files are formatted so that
48  // each row contains the same number of columns.
49  std::stringstream line_stream(line_);
50 
51  std::string field;
52 
53  while ( std::getline(line_stream, field, separator_) ) {
54  ++nCols_;
55  }

58  ++nCols_;

63 }
